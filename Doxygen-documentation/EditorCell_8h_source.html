100 void UpdateSelectionString();
101 // The selection/cursor setters clamp their argument to a valid index into the
102 // current text via ClampToText(), so m_selectionStart / m_selectionEnd are
103 // always in [0, m_text.Length()] (e.g. CursorMove(-n) past the start clamps to
104 // 0 instead of wrapping a size_t to a huge value). The getters below keep a
105 // defensive std::min as well, because the text can still shrink *after* a
106 // selection was set without the selection being updated.
107 void SetSelection(size_t start, size_t end){m_selectionStart = ClampToText(start); m_selectionEnd = ClampToText(end); UpdateSelectionString();}
108 bool SelectionActive() const {return m_selectionStart != m_selectionEnd;}
