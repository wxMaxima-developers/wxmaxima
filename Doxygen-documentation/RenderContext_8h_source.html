146 template <typename Report> void ReportMultipleRedraws(Report report)
147 {
148 if (!m_cellRedrawTrace)
149 return;
150 std::sort(m_cellRedrawTrace->begin(), m_cellRedrawTrace->end());
151 size_t counter = 0;
152 const Cell *prev = nullptr;
153 for (const Cell *cell : *m_cellRedrawTrace) {
154 if (prev != cell) {
155 if (counter > 1)
156 report(counter, prev);
157 prev = cell;
158 counter = 1;
159 } else
160 ++counter;
161 }
162 if (counter > 1)
163 report(counter, prev);
164 }

196 }
198 bool IsLayoutCancelled() const
199 {
200 if (m_layoutCancelled.load(std::memory_order_relaxed))
201 return true;
202 if (m_layoutDeadlineActive &&
203 std::chrono::steady_clock::now() >= m_layoutDeadline) {
204 m_layoutCancelled.store(true, std::memory_order_relaxed);
205 return true;
206 }
207 return false;
208 }
