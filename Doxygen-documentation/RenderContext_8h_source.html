133 bool IsLayoutCancelled() const
134 {
135 if (m_layoutCancelled.load(std::memory_order_relaxed))
136 return true;
137 if (m_layoutDeadlineActive &&
138 std::chrono::steady_clock::now() >= m_layoutDeadline) {
139 m_layoutCancelled.store(true, std::memory_order_relaxed);
140 return true;
141 }
142 return false;
143 }
