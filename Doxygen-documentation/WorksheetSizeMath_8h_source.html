69inline int ComputeWorksheetContentHeight(
70 const std::vector<TrailingGroupGeometry> &trailing, int groupSkip,
71 int baseIndent) {
72 int extraHeight = 0;
73 for (const TrailingGroupGeometry &cell : trailing) {
74 if (cell.sizeIsStale && cell.currentY >= 0)
75 // The anchor: its top position is trusted; add its drop and everything
76 // accumulated below it.
77 return cell.currentY + cell.maxDrop + extraHeight;
78 if (cell.sizeIsStale)
79 // Stale and not yet positioned (currentY < 0): count its own extent.
80 extraHeight += cell.maxDrop + groupSkip;
81 else
82 // Not stale but still unpositioned mid-layout: reserve a nominal height.
83 extraHeight += groupSkip + 20;
84 }
85 return baseIndent + extraHeight;
86}

118inline WorksheetVirtualSize ComputeWorksheetVirtualSize(bool hasTree,
119 int maxPointWidth,
120 int maxPointHeight,
121 int clientHeight,
122 int currentScrollPixelY) {
123 WorksheetVirtualSize result;
124 int scrollUnitBasisHeight = 40;
125 if (hasTree) {
126 result.width = maxPointWidth;
127 // Allow scrolling a little past the end of the document.
128 int height = maxPointHeight + clientHeight - clientHeight / 8;
129 // Keep a vertical scrollbar active.
130 result.height = std::max(clientHeight + 10, height);
131 // Never shrink below what the current scroll position requires.
132 result.height =
133 std::max(result.height, currentScrollPixelY + clientHeight + 10);
134 scrollUnitBasisHeight = clientHeight;
135 }
136 result.scrollUnit = std::max(scrollUnitBasisHeight / 30, 10);
137 return result;
138}

188inline void ApplyWorksheetVirtualSize(WorksheetView &view, bool hasTree,
189 int maxWidth, int maxHeight,
190 WorksheetVirtualSizeCache &cache,
191 int &scrollUnit) {
192 int clientWidth, clientHeight;
193 view.GetViewClientSize(&clientWidth, &clientHeight);
194
195 int currentScrollPixelY = 0;
196 if (hasTree)
197 currentScrollPixelY = view.GetViewScrollUnitY() * scrollUnit;
198
199 const WorksheetVirtualSize vs = ComputeWorksheetVirtualSize(
200 hasTree, maxWidth, maxHeight, clientHeight, currentScrollPixelY);
201
202 if ((cache.lastWidth != vs.width || cache.lastHeight != vs.height) &&
203 vs.height > 0) {
204 cache.lastWidth = vs.width;
205 cache.lastHeight = vs.height;
206 view.SetViewVirtualSize(vs.width, vs.height);
207 scrollUnit = vs.scrollUnit;
208 view.SetViewScrollRate(vs.scrollUnit);
209 }
210}
