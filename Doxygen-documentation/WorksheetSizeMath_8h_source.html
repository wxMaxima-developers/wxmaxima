67inline WorksheetVirtualSize ComputeWorksheetVirtualSize(bool hasTree,
68 int maxPointWidth,
69 int maxPointHeight,
70 int clientHeight,
71 int currentScrollPixelY) {
72 WorksheetVirtualSize result;
73 int scrollUnitBasisHeight = 40;
74 if (hasTree) {
75 result.width = maxPointWidth;
76 // Allow scrolling a little past the end of the document.
77 int height = maxPointHeight + clientHeight - clientHeight / 8;
78 // Keep a vertical scrollbar active.
79 result.height = std::max(clientHeight + 10, height);
80 // Never shrink below what the current scroll position requires.
81 result.height =
82 std::max(result.height, currentScrollPixelY + clientHeight + 10);
83 scrollUnitBasisHeight = clientHeight;
84 }
85 result.scrollUnit = std::max(scrollUnitBasisHeight / 30, 10);
86 return result;
87}
