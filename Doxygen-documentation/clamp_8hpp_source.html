30template <class T>
31constexpr const T &clamp(const T &val, const T &min, const T &max)
32{
33 assert(!(max < min));
34 return (val < min) ? min : (max < val) ? max : val;
35}
36
37template <class T, class Compare>
38constexpr const T &clamp(const T &val, const T &min, const T &max, Compare comp)
39{
40 assert(!comp(max, min));
41 return comp(val, min) ? min : comp(max, val) ? max : val;
42}
